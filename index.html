app = Dragonfly[:images].configure do |c|
  c.datastore =
    Dragonfly::DataStorage::S3DataStore.new

  c.datastore.configure do |d|
    # ... S3 configuration ...
  end

  c.define_url do |app, job, opts|
    Thumb.url_for_job(job)
    # Thumb also takes care of
    # S3 expiry.
  end
end
